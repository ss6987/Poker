    private static void createPlayer() {
        try {
            int port = 5001;
            ServerSocket ss = new ServerSocket(port);
            int req = 0;
            int playerNumber = 0;
            while (req == 0 && playerNumber < 5) {
                conect.add(new CommunicationSet(ss.accept()));
                player.add(new PlayerType(deck));
                System.out.println("接続");
                req = conect.get(playerNumber).getOis().readInt();
                conect.get(playerNumber).authorization(false);
                playerNumber = playerNumber + 1;
            }
            for(int k = 0;k < conect.size();k++){
                conect.get(k).authorization(true);
            }
        } catch (Exception e) {
            System.out.println("Exception:" + e);
        }
    }
